# The Flutter tooling requires that developers have CMake 3.10 or later
# installed. You should not increase this version, as doing so will cause
# the plugin to fail to compile for some customers of the plugin.
cmake_minimum_required(VERSION 3.10)

project(veil_light_plugin_library VERSION 0.0.1 LANGUAGES C)

add_library(veil_light_plugin SHARED
  "veil_light_plugin.c"
)

set_target_properties(veil_light_plugin PROPERTIES
  PUBLIC_HEADER veil_light_plugin.h
  OUTPUT_NAME "veil_light_plugin"
)

target_compile_definitions(veil_light_plugin PUBLIC DART_SHARED_LIB)
